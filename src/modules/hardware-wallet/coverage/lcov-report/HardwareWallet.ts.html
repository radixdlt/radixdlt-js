    errorHandler: (returnCode: number) => { returnCode: number, error: Error },
    responseHandler: (response: Buffer) => any,

    data: Buffer = Buffer.alloc(0),

        return responseHandler(await device.send(cla, ins, p1, p2, data))

    }
}
